from geopy.geocoders import Nominatim
geolocator=Nominatim(user_agent="app")
lat = []
lon = []

location = df2['Country']

for l in location:
    tmp = geolocator.geocode(l)
    if tmp is None:
        lat.append(np.nan)
        lon.append(np.nan)
    else:
        lat.append(tmp.latitude)
        lon.append(tmp.longitude)
        
df2['lat'] = lat
df2['lon'] = lon
